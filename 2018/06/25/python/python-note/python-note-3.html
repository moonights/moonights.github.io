python 逻辑

for循环

循环字符串

>>> for eachLetter in "Names":
...     print('letter:',eachLetter)
...
letter: N
letter: a
letter: m
letter: e
letter: s

range函数

>>> for i in range(1,10,2):
...     print(i)
...
1
3
5
7
9

循环列表

>>> for eachname in ['walter','nicole']:
...     print(eachname)
...
walter
nicole

>>> for i,j in enumerate(['b','a','c']):
...     print(i,j)
...
0 b
1 a
2 c

循环字典

>>> for k,v in {'a':1,'b':'c'}.items():
...     print(k,v)
...
a 1
b c

while语句

条件判断

>>> count = 0
>>> while count<4:
...     print(count)
...     count += 1
...
0
1
2
3

无限循环

>>> while True:
...     print(“endless loop”)
...

if语句

if…elif…else

if user_case == “create”:
    action = “create item”
elif user_case == 'delete':
    action = “delete item”
else:
    action = “invalid choice..”

缩进对于if语句格外重要

if user_case == “create”:
    if time_now < 5:
        action = “create item”
else:
    action = “do nothing”

三元表达式

>>> x = 5
>>> y = 4
>>> smaller = x if x<y else y
>>> smaller
4

break….pass….continue

for i in range(10):
    if i <5:
        pass
    elif i == 7:
        continue
    elif i == 8:
        print(i)
    else:
        break
